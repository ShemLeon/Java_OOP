package Exercices.EX1.Mivhan.E1_2025;



public class E1_Sheela2 {
    public static void main(String[] args) {
        A a1 = new C();
        // Здесь создается объект класса C
        // Переменной типа A присваивается ссылка на объект класса C
        // Это возможно, если класс C является наследником класса A
        // Приведение типов происходит автоматически (восходящее преобразование, upcasting)
        
        B b1 = (B)(new A());
        // Создается объект класса A
        // Затем происходит явное приведение типа к классу B с помощью (B)
        // Это нисходящее преобразование (downcasting)
        // Такое преобразование может вызвать ClassCastException, 
        //если класс B не является наследником класса A
        
        B b2 = new D();
        // Создается объект класса D
        // Переменной типа B присваивается ссылка на объект класса D
        // Это возможно, если класс D является наследником класса B
        // Приведение типов происходит автоматически (восходящее преобразование, upcasting)

    }
}
